**Goal:** Add ARM64 (AArch64) platform support to the LightLang compiler's code generation backend, enabling compilation for ARM-based systems like Apple Silicon, Linux ARM servers, and embedded devices.

**Architecture:** Create a new `ARM64Generator` struct implementing the `CodeGenerator` trait, following the same pattern as `X86_64Generator`. Map IR instructions to ARM64 assembly using AArch64 instruction set and ARM64 calling convention (AAPCS64). Generate position-independent code with proper stack management.

**Tech Stack:** Rust, ARM64 assembly, AAPCS64 calling convention, Linux/macOS ARM64 system calls

```rust
// src/codegen/arm64.rs
use crate::codegen::CodeGenerator;
use crate::ir::{IRFunction, IRInstruction, IRModule, IRType, IRValue};
use std::collections::HashMap;

pub struct ARM64Generator {
    var_map: HashMap<String, i32>,
    stack_size: i32,
}

impl ARM64Generator {
    pub fn new() -> Self {
        ARM64Generator {
            var_map: HashMap::new(),
            stack_size: 0,
        }
    }
    
    fn alloc_stack(&mut self, name: String) -> i32 {
        self.stack_size += 16; // ARM64 requires 16-byte alignment
        let offset = self.stack_size;
        self.var_map.insert(name, offset);
        offset
    }
    
    fn get_offset(&self, name: &str) -> Option<i32> {
        self.var_map.get(name).copied()
    }
}
```

**Step 4: Implement CodeGenerator trait**

```rust
impl CodeGenerator for ARM64Generator {
    fn generate(&mut self, module: &IRModule) -> Result<String, String> {
        let mut output = String::new();
        
        output.push_str(".section .text\n");
        output.push_str(".globl _start\n\n");
        
        for func in &module.functions {
            output.push_str(&self.generate_function(func));
        }
        
        output.push_str("_start:\n");
        output.push_str("    bl main\n");
        output.push_str("    mov x0, x0\n");
        output.push_str("    mov x8, #93\n"); // Linux exit syscall
        output.push_str("    svc #0\n");
        
        Ok(output)
    }
}
```

```rust
impl ARM64Generator {
    fn generate_function(&mut self, func: &IRFunction) -> String {
        let mut output = String::new();
        
        output.push_str(&format!("{}:\n", func.name));
        output.push_str("    stp x29, x30, [sp, #-16]!\n");
        output.push_str("    mov x29, sp\n");

        output.push_str("    ldp x29, x30, [sp], #16\n");
```

```rust
IRInstruction::Call {
    result,
    function,
    args,
} => {
    let mut code = String::new();
    
    // ARM64 calling convention: x0-x7 for arguments
    let arg_regs = ["x0", "x1", "x2", "x3", "x4", "x5", "x6", "x7"];
    
    for (i, arg) in args.iter().enumerate() {
        if i < arg_regs.len() {
            match arg {
                IRValue::Const(n) => {
                    code.push_str(&format!("    mov {}, #{}\n", arg_regs[i], n));
                }
                IRValue::Var(name) => {
                    if let Some(offset) = self.get_offset(name) {
                        code.push_str(&format!("    ldr {}, [x29, #-{}]\n", arg_regs[i], offset));
                    }
                }
            }
        }
    }
    
    code.push_str(&format!("    bl {}\n", function));
    
    if let Some(res) = result {
        self.alloc_stack(res.clone());
        if let Some(res_offset) = self.get_offset(res) {
            code.push_str(&format!("    str x0, [x29, #-{}]\n", res_offset));
        }
    }
    
    code
}
```

**Architecture Notes:**
- ARM64 uses 16-byte stack alignment (vs 8-byte for x86-64)
- System calls use `svc #0` with syscall number in x8
- AAPCS64 calling convention: x0-x7 for arguments,- Conditional execution uses `cset` instruction
- Division uses `sdiv` (signed division)
